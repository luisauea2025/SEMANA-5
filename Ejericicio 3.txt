using System;
using System.Collections.Generic;
using System.Linq;

namespace TareaNumeros
{
    class Program
    {
        static void Main(string[] args)
        {
            // 1. Crear y almacenar los números del 1 al 10 en una lista
            List<int> numeros = new List<int>();
            for (int i = 1; i <= 10; i++)
            {
                numeros.Add(i);
            }

            // 2. Invertir el orden de la lista
            numeros.Reverse();

            // 3. Mostrar por pantalla en orden inverso separados por comas
            // string.Join une los elementos con el separador indicado
            string resultado = string.Join(", ", numeros);
            
            Console.WriteLine("Números del 1 al 10 en orden inverso:");
            Console.WriteLine(resultado);

            // Pausa para ver el resultado en la terminal
            Console.WriteLine("\nPresione cualquier tecla para salir...");
            Console.ReadKey();
        } // Cierre del método Main
    } // Cierre de la clase Program
} // Cierre del namespace
